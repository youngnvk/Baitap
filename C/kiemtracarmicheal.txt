int k1[10000];
int b1[10000];

int gcd(int a,int b)
{
    while(b!=0)
    {
        int r=a%b;
        a=b;
        b=r;
    }
    return a;
}
int Bin(int n)
{
    int dem=0;
    while(n!=0)
    {
        int r=n%2;
        k1[dem]=r;
        dem++;
        n=n/2;
    }
    return dem;
}

int binhphuonglap(int a,int k,int n)// tÃ¬m mod
{
    int dem=Bin(k);
    int b=1;
    if(k==0)
    {
        return b;
    }
    int A=a;
    if(k1[0]==1)
    {
        b=a;
    }
    for(int i=1;i<=dem-1;i++)
    {
        A=(A*A)%n;
        if(k1[i]==1)
        {
            b=(A*b)%n;
        }
    }
    return b;
}


int kiemtrasocarmichael(int n)
{
    for(int i=2;i<n;i++)
    {
        if(gcd(n,i)==1)
        {
            if(binhphuonglap(i,n-1,n)!=1)
            {
                return 0;
            }
        }
    }
    return 1;

}
