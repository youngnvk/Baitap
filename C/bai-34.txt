
#include<stdio.h>

int k1[10000];

int Bin(int n)
{
    int dem=0;
    while(n!=0)
    {
        int r=n%2;
        k1[dem]=r;
        dem++;
        n=n/2;
    }
    return dem;
}

int binhphuonglap(int a,int k,int n)// t√¨m mod
{
    int dem=Bin(k);
    int b=1;
    if(k==0)
    {
        return b;
    }
    int A=a;
    if(k1[0]==1)
    {
        b=a;
    }
    for(int i=1;i<=dem-1;i++)
    {
        A=(A*A)%n;
        if(k1[i]==1)
        {
            b=(A*b)%n;
        }
    }
    return b;
}

int kiemTraFermat(int n,int t)
{
    for(int i=1;i<=t;i++)
    {
        int a=2+rand()%(n-2-2+1);
        int b=binhphuonglap(a,n-1,n);
        if(b!=1)
        {
            return false;
        }

    }
    return true;
}
